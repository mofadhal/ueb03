/**
 * Eine einfache Artikelklasse. Sie wurde wegen der fuenften Aufgabe um das Attribut preis und die entsprechenden Methoden erweitert.
 * Auch bei den Konstruktoren und bei der toString()-Methode wird das neue Attribut beruecksichtigt. Eine equals()-Methode wurde
 * eingefuegt, um Vergleiche von Artikeln zuzulassen.
 * @author Christopher
 *
 */
public class Artikel {
	private static final String FALSCHE_ARTIKELNUMMER = "Die Artikelnummer muss zwischen 999 und 10000 liegen!";
	private static final String LEERE_BEZEICHNUNG = "Die Bezeichnung darf nicht leer bleiben!";
	private static final String NIEDRIGER_BESTAND = "Der Bestand muss >= 0 sein!";
	private static final String MENGE_KLEINER_NULL = "Die Menge muss > 0 sein!";
	private static final String BESTAND_KLEINER_NULL = "Der neue Bestand muss > 0 sein!";
	/**
	 * Der Anfangsbestand, mit dem neue Artikel initialisiert werden.
	 */
	private static final int ANFANGS_BESTAND = 0;
	/**
	 * Die Artikelnummer muss zwischen 999 und 10000 liegen.
	 */
    private int artikelNr;
    /**
     * Die Bezeichnung des Artikels muss eingegeben werden.
     */
    private String bezeichnung;
    /**
     * Der Lagerbestand des Artikels.
     */
    private int bestand;

    /**
     * Konstruktor, der einen Artikel initialisiert. Die Uebergabeparameter werden geprueft. 
     * @param artikelNr Die Artikelnummer muss vierstellig sein.
     * @param bezeichnung Die Artikelbezeichnung muss einen Text enthalten.
     * @param bestand Der Bestand des Artikels muss positiv oder 0 sein.
     * @param preis Der Preis des Artikels muss groesser als 0 sein. 
     */
    public Artikel(int artikelNr, String bezeichnung, int bestand) {
        assert (artikelNr > 999 && artikelNr < 10000): FALSCHE_ARTIKELNUMMER;
        assert (bezeichnung != null && 
        		!bezeichnung.trim().isEmpty()): LEERE_BEZEICHNUNG;
        assert (bestand >= ANFANGS_BESTAND): NIEDRIGER_BESTAND; 
        
        this.artikelNr = artikelNr;
        this.bezeichnung = bezeichnung.trim();
        this.bestand = bestand;
    }
    
    /**
     * Konstruktor, der einen Artikel mit einem Anfangsbestand initialisiert.
     * @param artikelNr Die Artikelnummer.
     * @param bezeichnung Die Artikelbezeichnung.
     * @param preis Der Preis des Artikels.
     */
    public Artikel(int artikelNr, String bezeichnung) {
    	this(artikelNr, bezeichnung, ANFANGS_BESTAND);
    }
    /**
     * Bucht einen Zugang zum Artikelbestand. Dabei muss die zugebuchte Menge groesser als 0 sein.
     * @param menge Die Menge, die dazu gebucht wird.
     */
    public void bucheZugang(int menge) {
        assert (menge > 0): MENGE_KLEINER_NULL;
        this.bestand = bestand + menge;
    }
    
    /**
     * Bucht einen Abgang vom Artikelbestand. Die abgebuchte Menge muss groesser als 0 sein. Ausserdem darf der 
     * Bestand durch die Abbuchung nicht unter 0 fallen.
     * @param menge Die abzubuchende Menge.
     */
    public void bucheAbgang(int menge) {
        assert (menge > 0): MENGE_KLEINER_NULL;
        assert (this.bestand > menge): BESTAND_KLEINER_NULL;
        this.bestand = this.bestand - menge;
    }
    
    /**
     * Wandelt den Artikel in einen String um.
     * @return Die String-Repraesentation des Artikels.
     */
    public String toString() {
    	return "Artikel: " + artikelNr + " Bezeichnung: " + bezeichnung + " Bestand: " + bestand;
    }
    
	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + artikelNr;
		result = prime * result + ((bezeichnung == null) ? 0 : bezeichnung.hashCode());
		return result;
	}

	@Override
	public boolean equals(Object obj) {
		if (this == obj)
			return true;
		if (obj == null)
			return false;
		if (getClass() != obj.getClass())
			return false;
		Artikel other = (Artikel) obj;
		if (artikelNr != other.artikelNr)
			return false;
		if (bezeichnung == null) {
			if (other.bezeichnung != null)
				return false;
		} else if (!bezeichnung.equals(other.bezeichnung))
			return false;
		return true;
	}

	public String getBezeichnung() {
		return bezeichnung;
	}

	public void setBezeichnung(String bezeichnung) {
        assert (bezeichnung != null && !bezeichnung.trim().isEmpty()): LEERE_BEZEICHNUNG;
		this.bezeichnung = bezeichnung.trim();
	}

	public int getArtikelNr() {
		return artikelNr;
	}

	public int getBestand() {
		return bestand;
	}
}
