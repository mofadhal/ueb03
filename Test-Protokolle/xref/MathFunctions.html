9   public class MathFunctions {
10      private static final long MIN_ISBN = 100000000;
11      private static final long MAX_ISBN = 1000000000;
12  
13      /**
14       * Vorschleife zur berechnung der Teilersumme mit mod 2
15       *
16       * @param zahl if Bedingung zur Überprüfung der zahl ob sie eine Teilersummme ist
17       * @return summe Ausgabe der Summe
18       */
19  
20  
21      static long berechneTeilersumme(long zahl) {
22          int Teilersumme = 0;
23          if(zahl > 0){
24          for (int i = 1; i <= zahl; i++) {
25              if ((zahl % i) == 0) {
26                  Teilersumme = Teilersumme + i;
27              }
28          }
29         }else {
30             throw new IllegalArgumentException("Die Zahl muss positiv sein");
31          }
32          return Teilersumme;
33      }
34      
35      /**
36       * Vorschleife zur berechnung der pruefziffer 
37       *
38       * @param isbn: isbn is eine 9 stellige zahl und anhand diese zahl wird die 
39       *              pruef ziffer wie in blatt beschreibung berechnet
40       * @return Die berechnete prüfziffer als String
41       */
42  
43      static String berechneChecksummeIsbn(long isbn) {
44  
45          int checkDigit = 0;
46          if (isbn >= MIN_ISBN && isbn < MAX_ISBN) {
47              for (int j = 9; j >= 1; j--) {
48                  checkDigit += j * (isbn % 10);
49                  isbn /= 10;
50              }
51              checkDigit %= 11;
52          } else {
53              throw new IllegalArgumentException("isbn muss = 9 stillige positiv Nummern sein");
54          }
55          if (checkDigit == 10) {
56              return "X";
57          }
58          return " " + checkDigit;
59      }
